// Firebase Configuration Template
// 
// IMPORTANT: This file needs to be generated using FlutterFire CLI
// 
// Steps to generate firebase_options.dart:
//
// 1. Install FlutterFire CLI:
//    dart pub global activate flutterfire_cli
//
// 2. Create a Firebase project:
//    - Go to https://console.firebase.google.com/
//    - Click "Create a project"
//    - Enter project name: "botabaixo" (or your preferred name)
//    - Enable Google Analytics (optional)
//    - Click "Create project"
//
// 3. Enable Anonymous Authentication:
//    - In Firebase Console, go to "Authentication"
//    - Click "Get started"
//    - Go to "Sign-in method" tab
//    - Enable "Anonymous" provider
//
// 4. Create Firestore Database:
//    - In Firebase Console, go to "Firestore Database"
//    - Click "Create database"
//    - Choose "Start in test mode"
//    - Select a location close to your users
//    - Click "Enable"
//
// 5. Run FlutterFire configure:
//    flutterfire configure
//
//    This will generate lib/firebase_options.dart with your project config.
//
// 6. Add Android app:
//    - In Firebase Console, click "Add app" > Android
//    - Enter package name: com.botabaixo.botabaixo
//    - Download google-services.json
//    - Place it in android/app/google-services.json
//
// 7. Add iOS app:
//    - In Firebase Console, click "Add app" > iOS
//    - Enter bundle ID: com.botabaixo.botabaixo
//    - Download GoogleService-Info.plist
//    - Place it in ios/Runner/GoogleService-Info.plist
//
// 8. Run the app:
//    flutter run
//
// The app will work offline if Firebase is not configured.
// Firebase is optional - all data is stored locally with Hive.
